233 {

263 private void UpdateVisualization(float time)
264 {
265 for (int i = 0; i < currentSession.recordedObjects.Count; i++)
266 {
267 if (i >= visualObjects.Count) continue;
268
269 var recordedObject = currentSession.recordedObjects[i];
270 var visualObject = visualObjects[i];
271
272 if (recordedObject.recordedFrames.Count == 0) continue;
273
274 // Find the appropriate frames to interpolate between
275 TransformData prevFrame = recordedObject.recordedFrames[0];
276 TransformData nextFrame = recordedObject.recordedFrames[0];
277
278 for (int j = 0; j < recordedObject.recordedFrames.Count - 1; j++)
279 {
280 if (recordedObject.recordedFrames[j].timestamp <= time &&
281 recordedObject.recordedFrames[j + 1].timestamp >= time)
282 {
283 prevFrame = recordedObject.recordedFrames[j];
284 nextFrame = recordedObject.recordedFrames[j + 1];
285 break;
286 }
287 }
288
289 // If we're beyond the last frame, use the last one
290 if (time > recordedObject.recordedFrames[recordedObject.recordedFrames.Count - 1].timestamp)
291 {
292 prevFrame = nextFrame = recordedObject.recordedFrames[recordedObject.recordedFrames.Count - 1];
293 }
294
295 // Interpolate between frames
296 float t = Mathf.InverseLerp(prevFrame.timestamp, nextFrame.timestamp, time);
297
298 // Apply interpolated transform
299 visualObject.transform.position = Vector3.Lerp(prevFrame.position, nextFrame.position, t);
300 visualObject.transform.rotation = Quaternion.Slerp(prevFrame.rotation, nextFrame.rotation, t);
301 }

311 private void CreateVisualObjects()
312 {
313 if (currentSession == null) return;
314
315 // Create a parent object for all replay visualizations
316 replayParentObject = new GameObject("Replay_Visualization");
317 replayParentObject.hideFlags = HideFlags.DontSave;
318
319 foreach (var recordedObject in currentSession.recordedObjects)
320 {
321 GameObject visualObject = null;
322
323 // Try to find the original object in the scene to use its shape
324 GameObject originalObject = GameObject.Find(recordedObject.objectName);
325
326 if (originalObject != null)
327 {
328 // Create a clone of the original object
329 visualObject = GameObject.Instantiate(originalObject);
330 visualObject.name = $"Replay_{recordedObject.objectName}";
331
332 // Reset the transform before applying recorded data
333 // This is crucial for objects with parent-based scale modifications
334 visualObject.transform.SetParent(replayParentObject.transform, false);
335
336 // Reset local scale to remove parent scale influence
337 visualObject.transform.localScale = Vector3.one;
338
339 // Calculate the global scale from the original object by traversing its hierarchy
340 Vector3 globalScale = originalObject.transform.lossyScale;
341
342 // Apply the global scale to our independent replay object
343 visualObject.transform.localScale = globalScale;
344
345 // Disable any scripts or components that might interfere
346 MonoBehaviour[] scripts = visualObject.GetComponents<MonoBehaviour>();
347 foreach (MonoBehaviour script in scripts)
348 {
349 script.enabled = false;
350 }
351
352 // Remove rigidbody to prevent physics interactions
353 Rigidbody rb = visualObject.GetComponent<Rigidbody>();
354 if (rb != null)
355 {
356 Object.DestroyImmediate(rb);
357 }
358
359 // Remove colliders to prevent physics interactions
360 Collider[] colliders = visualObject.GetComponentsInChildren<Collider>();
361 foreach (Collider collider in colliders)
362 {
363 Object.DestroyImmediate(collider);
364 }
365
366 // Apply semi-transparent material to all renderers
367 Renderer[] renderers = visualObject.GetComponentsInChildren<Renderer>();
368 foreach (Renderer renderer in renderers)
369 {
370 // Create a new material instance based on the original
371 Material originalMat = renderer.sharedMaterial;
372 if (originalMat != null)
373 {
374 Material newMat = new Material(originalMat);
375
376 // Set appropriate color with transparency
377 Color color = recordedObject.isPlayer ? playerColor : normalObjectColor;
378 newMat.color = color;
379
380 // Apply the material
381 renderer.sharedMaterial = newMat;
382 }
383 }
384 }
385 else
386 {
387 // Fallback to primitives if original object cannot be found
388 if (recordedObject.isPlayer)
389 {
390 // Create a capsule for players
391 visualObject = GameObject.CreatePrimitive(PrimitiveType.Capsule);
392 visualObject.name = $"Replay_{recordedObject.objectName}";
393 visualObject.transform.localScale = new Vector3(1f, 1f, 1f);

403 }
