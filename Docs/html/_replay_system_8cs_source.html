335 public void StartRecording(List<Rigidbody> playerRigidbodies, List<Rigidbody> normalRigidbodies)
336 {

384 public ReplaySession SaveCurrentRecording(string sessionName = "")
385 {
386 if (currentRecordingData.Count == 0)
387 {
388 Debug.LogWarning("No recording data to save");
389
390 return null;
391 }
392
393 var session = new ReplaySession(sessionName);
394 session.duration = RecordingDuration; // This will now use the correct duration
395
396 // Deep copy the recording data
397 foreach (var recordingData in currentRecordingData)
398 {
399 var sessionData = new RigidbodyRecordingData(recordingData.rigidbody, recordingData.isPlayer);
400 sessionData.objectName = recordingData.objectName;
401 sessionData.recordedFrames = new List<TransformData>(recordingData.recordedFrames);
402 session.recordedObjects.Add(sessionData);
403 }
404
405 // Add to session manager
406 ReplaySessionManager.Instance.AddSession(session);
407
408 Debug.Log($"Saved replay session: {session.name} with duration {session.duration:F2}s and {session.recordedObjects.Count} objects");
409
410 return session;

441 private void Update()
442 {
443 if (!isRecording || !Application.isPlaying) return;
444
445 float currentTime = Time.time - recordingStartTime;
446
447 // Record player objects at higher frequency
448 if (currentTime - lastPlayerRecordTime >= playerRecordingInterval)
449 {
450 foreach (var data in currentRecordingData)
451 {
452 if (data.isPlayer && data.rigidbody != null)
453 {
454 data.AddFrame(currentTime);
455 }
456 }
457
458 lastPlayerRecordTime = currentTime;
459 }
460
461 // Record normal objects at lower frequency
462 if (currentTime - lastNormalRecordTime >= normalRecordingInterval)
463 {
464 foreach (var data in currentRecordingData)
465 {
466 if (!data.isPlayer && data.rigidbody != null)
467 {
468 data.AddFrame(currentTime);
469 }
470 }
471
472 lastNormalRecordTime = currentTime;
473 }
